一、作品概述

"爱即守护"是一款基于微信小程序的居家语音关怀平台。老人通过 ASR Pro 离线语音设备喊出关键词，小程序实时接收告警和留言。DeepSeek 大模型驱动 AI 陪聊，百度语音识别合成实现全语音对话。家人端提供仪表盘、告警管理、留言板、用药提醒等功能。

核心价值：老人喊一嗓子，家人小程序收到；AI 陪老人聊天解闷。零触屏语音交互，WiFi断线也能本地安抚。

3.1 语音SOS全链路

老人对着ASR Pro喊"救命"→ 语音识别模块输出关键词 → ESP32通过WiFi上传 → 云函数接收翻译 → alerts集合写入 → 家人端小程序显示红色告警。

ESP32固件（__WiFi_____.ino）— 接收ASR Pro文字并上传云端

// 爱即守护 - ESP32 固件 v3.3（支持多命令词）
#include <WiFi.h>
#include <HTTPClient.h>
#include <HardwareSerial.h>

#define WIFI_SSID       "su"
#define WIFI_PASSWORD   "subowen123"
#define DEVICE_ID       "001"
#define CLOUD_FUNC_URL  "https://cloud2-d2g7yeednd8ac1836-1433086696.ap-shanghai.app.tcloudbase.com/deviceDataIngest"
#define ASR_RX_PIN      16
#define REPEAT_INTERVAL 10000

HardwareSerial asrSerial(2);
String lastText = "";
unsigned long lastUploadTime = 0, lastReconnectAttempt = 0;

bool isEmergency(String text) {
    return (text == "救命") || (text == "我摔倒了");
}

void uploadToCloud(String text, bool emergency) {
    if (WiFi.status() != WL_CONNECTED) { return; }
    HTTPClient http;
    http.begin(CLOUD_FUNC_URL);
    http.addHeader("Content-Type", "application/json");
    String json = "{\"deviceId\":\"" + String(DEVICE_ID) + "\",\"type\":\""
                + String(emergency ? "voice_sos" : "voice_msg") + "\",\"text\":\"" + text + "\"}";
    int httpCode = http.POST(json);
    http.end();
}

void setup() {
    Serial.begin(115200);
    asrSerial.begin(115200, SERIAL_8N1, ASR_RX_PIN, -1);
    WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
    int tries = 0;
    while (WiFi.status() != WL_CONNECTED && tries++ < 30) { delay(500); }
}

void loop() {
    static String buffer = "";
    while (asrSerial.available()) {
        char c = asrSerial.read();
        if (c == '\n' || c == '\r') {
            if (buffer.length() > 0) {
                String currentText = buffer; buffer = "";
                unsigned long now = millis();
                if (currentText != lastText || (now - lastUploadTime) > REPEAT_INTERVAL) {
                    lastText = currentText;
                    lastUploadTime = now;
                    uploadToCloud(currentText, isEmergency(currentText));
                }
            }
        } else { buffer += c; if (buffer.length() > 200) buffer = ""; }
    }
    if (WiFi.status() != WL_CONNECTED && millis() - lastReconnectAttempt > 10000) {
        lastReconnectAttempt = millis();
        WiFi.reconnect();
    }
    delay(10);
}

云函数 deviceDataIngest/index.js — 接收数据、翻译关键词、写入数据库

const cloud = require('wx-server-sdk')
cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV })
const db = cloud.database()

exports.main = async (event) => {
  let params = event
  if (typeof event.body === 'string') {
    try { params = JSON.parse(event.body) } catch(e) {}
  }
  const { deviceId, type } = params
  const rawText = params.text || ''
  const map = { help:'救命', fell:'我摔倒了', ok:'我很好', miss:'想你们了', pill:'该吃药了' }
  const text = map[rawText] || rawText
  const ts = new Date()

  try {
    const did = deviceId || '001'
    // 设备状态（自动创建）
    const dev = await db.collection('devices').where({ deviceId: did }).get()
    if (dev.data.length > 0) {
      await db.collection('devices').doc(dev.data[0]._id).update({ data: { status: 'online', lastSeen: ts } })
    } else {
      await db.collection('devices').add({ data: { deviceId: did, status: 'online', lastSeen: ts } })
    }
    // 服务端判断SOS（不依赖ESP32的type字段）
    const isSOS = type === 'voice_sos' || (text && (
      text == 'help' || text == 'fell' ||
      text.indexOf('救命') >= 0 || text.indexOf('摔倒') >= 0 || text.indexOf('救我') >= 0))

    if (isSOS && text) {
      await db.collection('alerts').add({ data: { deviceId: did, type: 'voice_sos', level: 'critical', message: '语音求救: ' + text, handled: false, timestamp: ts } })
    }
    if (!isSOS && text) {
      await db.collection('messages').add({ data: { type: 'voice', content: text, fromRole: 'elderly', fromUser: '老人(语音)', createTime: ts } })
    }
    return { success: true, type: type || 'unknown' }
  } catch (err) {
    return { success: false, error: err.message }
  }
}

3.2 AI语音陪聊（pages/elderly/chat/chat.js）

按住🎤说话 → 百度语音识别(STT) → DeepSeek大模型生成回复 → 百度语音合成(TTS)朗读。全程语音，老人不用打字。文字输入同样支持。

const DEEPSEEK_KEY = 'sk-xxx'
const BAIDU_KEY = 'xxx'
const BAIDU_SECRET = 'xxx'
let baiduToken = '', baiduTokenTime = 0

Page({
  data: { messages: [], inputText: '', typing: false, listening: false },

  onLoad() {
    this.setData({ messages: [{ role: 'bot', text: '您好！我是小安。打字或点🎤跟我语音对话~', time: this.now() }] })
  },

  // ===== 文字发送 =====
  onSend() {
    const text = this.data.inputText.trim()
    if (!text || this.data.typing) return
    this.setData({ inputText: '', typing: true })
    this.addMsg('user', text)
    this.chat(text)
  },

  quickChat(e) {
    const t = e.currentTarget.dataset.text
    if (typeof t !== 'string') return
    this.setData({ typing: true })
    this.addMsg('user', t); this.chat(t)
  },

  // ===== 语音输入（按住说话，松开发送）=====
  startVoice() {
    if (this._recording) return
    this._recording = true
    const rm = wx.getRecorderManager()
    this._rm = rm
    rm.onStop((res) => {
      this._recording = false; this._rm = null
      this.setData({ listening: false })
      if (res.tempFilePath) { wx.showLoading({ title: '识别中...' }); this.stt(res.tempFilePath) }
    })
    rm.onError((err) => {
      this._recording = false; this._rm = null
      wx.showToast({ title: '录音失败，请授权麦克风', icon: 'none' })
    })
    rm.start({ duration: 15000, sampleRate: 16000, numberOfChannels: 1, format: 'wav' })
  },

  stopVoice() {
    if (!this._recording) return
    this._rm && this._rm.stop()
  },

  // ===== 百度语音识别 =====
  stt(filePath) {
    const fs = wx.getFileSystemManager()
    const data = fs.readFileSync(filePath)
    this.baiduToken().then(token => {
      wx.request({
        url: 'https://vop.baidu.com/server_api', method: 'POST',
        data: { format: 'wav', rate: 16000, channel: 1, token, cuid: 'wx_miniprogram', speech: wx.arrayBufferToBase64(data), len: data.byteLength },
        success: (res) => {
          wx.hideLoading()
          if (res.data.err_no === 0 && res.data.result?.length > 0) {
            this.setData({ typing: true })
            this.addMsg('user', res.data.result[0])
            this.chat(res.data.result[0])
          } else { wx.showToast({ title: '没听清，再说一遍', icon: 'none' }) }
        }
      })
    })
  },

  // ===== DeepSeek AI 对话 =====
  chat(text) {
    wx.request({
      url: 'https://api.deepseek.com/v1/chat/completions', method: 'POST',
      header: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${DEEPSEEK_KEY}` },
      data: { model: 'deepseek-chat', messages: [{ role: 'user', content: '你是"小安"，温暖贴心的AI陪聊伙伴。\n老人说：' + text }], temperature: 0.85, max_tokens: 600 },
      timeout: 20000,
      success: (res) => {
        this.setData({ typing: false })
        if (res.statusCode === 200 && res.data?.choices) {
          const reply = res.data.choices[0].message.content.trim()
          this.addMsg('bot', reply)
          this.tts(reply)
        }
      }
    })
  },

  // ===== 百度TTS语音朗读 =====
  tts(text) {
    this.baiduToken().then(token => {
      const audio = wx.createInnerAudioContext()
      audio.src = `https://tsn.baidu.com/text2audio?tok=${token}&cuid=wx&ctp=1&lan=zh&spd=5&pit=5&vol=9&per=0&aue=3&tex=${encodeURIComponent(text.substring(0,200))}`
      audio.play()
    }).catch(() => {})
  },

  addMsg(role, text) { this.setData({ messages: [...this.data.messages, { role, text, time: this.now() }] }) },

  baiduToken() {
    return new Promise((resolve, reject) => {
      if (baiduToken && Date.now() - baiduTokenTime < 3600000) return resolve(baiduToken)
      wx.request({
        url: `https://aip.baidubce.com/oauth/2.0/token?grant_type=client_credentials&client_id=${BAIDU_KEY}&client_secret=${BAIDU_SECRET}`,
        success: (r) => { if (r.data.access_token) { baiduToken = r.data.access_token; baiduTokenTime = Date.now(); resolve(baiduToken) } else reject() }
      })
    })
  }
})

// 老人端及家人端公共逻辑
checkDevice() {
  wx.cloud.database().collection('devices').where({ deviceId: '001' }).limit(1).get()
    .then(res => {
      if (res.data.length > 0) {
        const d = res.data[0]

    })
}

3.6 告警系统

支持语音求救(voice_sos)、SOS按钮、跌倒检测三种告警类型。筛选、标记处理、一键清空已处理。本地即时更新UI无需等云端返回。

// 告警处理 — 本地先更新UI
handleAlert(e) {
  const id = e.currentTarget.dataset.id
  const alerts = this.data.alerts.map(a => a._id === id ? { ...a, handled: true } : a)
  this.setData({ alerts })  // 即时响应
  wx.cloud.database().collection('alerts').doc(id).update({ data: { handled: true } })
}

// 一键清空已处理
clearHandled() {
  const handled = this.data.alerts.filter(a => a.handled)
  const db = wx.cloud.database()
  const del = (i) => {
    if (i >= handled.length) { this.loadAlerts(); return }
    db.collection('alerts').doc(handled[i]._id).remove().then(() => del(i+1)).catch(() => del(i+1))
  }
  del(0)
}

3.7 AI日报（family/dashboard/dashboard.js）

统计当天语音消息数和告警数，调用DeepSeek生成口语化一句话摘要。

loadDailyReport() {
  const today = new Date(); today.setHours(0,0,0,0)
  const db = wx.cloud.database()
  db.collection('messages').where({ createTime: db.command.gte(today) }).count()
    .then(r1 => {
      db.collection('alerts').where({ timestamp: db.command.gte(today) }).count()
        .then(r2 => {
          const prompt = `根据今天数据生成2句日报：语音消息${r1.total}条，告警${r2.total}次。口语化，30字内。`
          wx.request({
            url: 'https://api.deepseek.com/v1/chat/completions', method: 'POST',
            header: { 'Authorization': 'Bearer sk-xxx' },
            data: { model: 'deepseek-chat', messages: [{ role: 'user', content: prompt }], temperature: 0.5, max_tokens: 100 },
            success: (r) => this.setData({ dailyReport: r.data.choices[0].message.content }),
            fail: () => this.setData({ dailyReport: `今日语音${r1.total}条，告警${r2.total}次。` })
          })
        })
    })
}

loadWeather() {
  wx.request({
    url: 'https://wttr.in?format=%C', timeout: 5000,
    success: (res) => {
      const w = (res.data||'').toLowerCase()
      let tip = ''
      if (w.includes('rain') || w.includes('drizzle')) tip = '🌧 今日有雨，地面湿滑，注意防滑'
      else if (w.includes('snow')) tip = '❄️ 今日有雪，谨慎外出'
      else if (w.includes('wind')) tip = '💨 今日风大，注意安全'
      else if (w.includes('sunny')||w.includes('clear')) tip = '☀️ 今日晴天，适合晒太阳'
      else tip = '🌤 注意天气变化'
      this.setData({ weatherTip: tip })
    }
  })
}

六、应用场景

场景一：语音求救 — 李奶奶在卫生间滑倒，无法移动。喊了一声"救命"，家里设备立刻回应"已通知家人"。女儿手机小程序弹出红色告警"🔊语音求救"。

场景二：日常报平安 — 张爷爷每天早上对着设备说"我很好"。设备回复"已告诉家人"。儿子上班路上打开小程序，留言板橙色消息"我很好"——父亲状态不错。

场景三：AI陪伴 — 王奶奶独居觉得闷，打开小程序按住说话："小安，今天有点闷"。小安语音回复："老人家别闷啦，我给您讲个笑话吧……"

场景四：远程关怀 — 赵女士在外地工作，在留言板写道"妈今天降温多穿点"并设了降压药提醒。母亲手机首页大字卡片显示留言，到点弹出用药提醒。

七、总结

"爱即守护"不是一套冰冷的监测系统。它是用AI和语音技术搭建的一座桥——老人不用学习复杂的操作，家人不用担忧错过重要的信息。老人端温暖简单，像一盏灯；家人端专业清晰，像一块仪表盘。我们相信科技的最终价值不是更复杂的算法，而是更简单的生活。
